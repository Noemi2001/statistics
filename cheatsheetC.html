        void SetQueueValue(Queue q, int oldValue, int newValue) // set a new value
        {
            int[] items = q.ToArray();
            for (int i = 0; i < items.Length; i++)
            {
                if (items[i] == oldValue)
                {
                    items[i] = newValue;
                    q.Clear();
                    foreach (var item in items)
                        q.Enqueue(item);
                    break;
                }
            }
        }

        void SetLinkedListValue(LinkedList list, int oldValue, int newValue) //set a new value
        {
            var currentNode = list.First;
            while (currentNode != null)
            {
                if (currentNode.Value == oldValue)
                {
                    currentNode.Value = newValue;
                    break;
                }
                currentNode = currentNode.Next;
            }
