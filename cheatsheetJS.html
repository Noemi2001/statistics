SORTED LIST

    
    class SortedList {
    constructor() {
        this.items = [];
    }

    insert(item) {  // Insert an element into the sorted list while maintaining the order.
        const index = this.findInsertIndex(item);
        this.items.splice(index, 0, item);
    }
    
    findInsertIndex(item) { // Find the index where the item should be inserted to maintain the sorted order.
        let index = 0;
        while (index < this.items.length && item > this.items[index]) {
        index++;
        }
        return index;
    }

    remove(item) { // Remove an item from the sorted list.
        const index = this.items.indexOf(item);
        if (index !== -1) {
        this.items.splice(index, 1);
        }
    }

    elementExists(sortedList, target) { // Check if an element exists in the sorted list.
        for (const item of sortedList.items) {
        if (item === target) {
            return true;
        }
        }
        return false;
    }

    isEmpty() {
        return this.items.length === 0; // Check if the sorted list is empty.
    }

    forEach(callback) { //loop
        for (const item of this.items) {
        callback(item);
        }
    }

    setValue(oldValue, newValue) {
        this.remove(oldValue);
        this.insert(newValue);
    }
    }

HASHSET

         
    class HashSet {
        constructor() {
          this.values = new Map();
        }

        add(value) { // add new element in the hashset
          this.values.set(value, true);
        }

        delete(value) { // remove an element from the hashset
          this.values.delete(value);
        }

        exist(value) { // Check if a specific value is in the hashset 
          if(this.values.has(value) != null){
            return true
          }
          return false
        }

        getValue(value) { // return a specific value is in the hashset.
          tmp = this.values.has(value);
          if(tmp != null){
            return tmp
          }
          return null
        }

        forEach(callback) { 
          for (const value of this.values.keys()) { //loop
            callback(value);
          }
        }

        setValue(oldValue, newValue) {
          if (this.exists(oldValue)) {
            this.delete(oldValue);
            this.add(newValue);
          }
        }
    }

SORTED SET

      
      class SortedSet {
        constructor() {
          this.data = [];
        }

        add(value) { // Add a value to the sorted set while maintaining the order.
          if (!this.has(value)) {
            this.data.push(value);
            this.data.sort((a, b) => a - b); // Sort the array in ascending order.
          }
        }

        delete(value) { // Remove a value from the sorted set.
          const index = this.data.indexOf(value);
          if (index !== -1) {
            this.data.splice(index, 1);
          }
        }

        exist(value) { // Check if a value is in the sorted set.
          if (this.data.includes(value)){
            return true;
          }
          return false;
        }

        getValue(value){ // return a value if exist in the sorted set.
          tmp = this.data.includes(value)
          if (tmp != null){
            return tmp;
          }
          return null;

        forEach(callback) { //loop
          for (const value of this.data) {
            callback(value);
          }
        }

        setValue(oldValue, newValue) {
          this.delete(oldValue);
          this.add(newValue);
        }
      }

STACK

        
      class Stack {
        constructor() {
          this.items = [];
        }
      
        push(element) { // add an element onto the stack
          this.items.push(element);
        }
      
        pop() { // remove an element from the stack and return it
          if (!this.isEmpty()) {
            return this.items.pop();
          }
        }
      
        get() { // Peek at the top element without removing it
          if (!this.isEmpty()) {
            return this.items[this.items.length - 1];
          }
        }
      
        isEmpty() { // Check if the stack is empty
          return this.items.length === 0;
        }
      
        forEach(callback) { //loop
          for (const element of this.items) {
            callback(element);
          }
        }
      
        setValue(oldValue, newValue) {
          if (this.isEmpty()) {
            return;
          }
          const index = this.items.lastIndexOf(oldValue);
          if (index !== -1) {
            this.items.splice(index, 1); // Remove the old value
            this.push(newValue); // Push the new value
          }
        }
      }

LINKED LIST

        
      class Node {
        constructor(data) {
          this.data = data;
          this.next = null;
        }
      }
      
      class LinkedList {
        constructor() {
          this.head = null;
        }
      
        append(data) { //insert a new element at the end of the linked list
          const newNode = new Node(data);
          if (!this.head) {
            this.head = newNode;
          } else {
            let current = this.head;
            while (current.next) {
              current = current.next;
            }
            current.next = newNode;
          }
        }
      
        add(data) { // Add a new element to the beginning of the list.
          const newNode = new Node(data);
          newNode.next = this.head;
          this.head = newNode;
        }
      
        delete(data) { // Delete the first occurrence of a node with the given data.
          if (!this.head) {
            return;
          }
      
          if (this.head.data === data) {
            this.head = this.head.next;
            return;
          }
      
          let current = this.head;
          while (current.next) {
            if (current.next.data === data) {
              current.next = current.next.next;
              return;
            }
            current = current.next;
          }
        }
      
        exist(data) { // Check if the linked list contains the specified data element.
          let current = this.head;
          while (current) {
            if (current.data === data) {
              return true;
            }
            current = current.next;
          }
          return false;
        }
      
        get(data) { // Return the first occurrence of the specified data element in the linked list.
          let current = this.head;
          while (current) {
            if (current.data === data) {
              return current.data;
            }
            current = current.next;
          }
          return null;
        }
      
        forEach(callback) { //loop
          let current = this.head;
          while (current) {
            callback(current.data);
            current = current.next;
          }
        }
      
        setValue(oldValue, newValue) {
          let current = this.head;
          while (current) {
            if (current.data === oldValue) {
              current.data = newValue; // Update the value
              break;
            }
            current = current.next;
          }
        }
      }
